context ContextClass
inv ConstraintName:
  self.status = StudentStatus::ENROLLED implies
let v : Integer =
      self.enrolments
          ->select(e | e.offering.semester = self.currentSemester)
          ->collect(e | e.course.credits)
          ->sum()
  in
      v <= self.maxCredits

context ContextClass
inv ConstraintName:
  let v : Integer =
      self.studyRecords
          ->select(e | e.semester = self.currentSemester)
          ->collect(e | e.self.studyHours)
          ->sum()
  in
      v >= 120

context ContextClass
inv ConstraintName:

          ->select(e | e.semester = self.currentSemester)
          ->collect(e | e.self.studyHours)
          ->sum()

      s = self.totalStudyHours

context ContextClass
inv ConstraintName:
  let s : Integer =
      self.enrolments
          ->select(e | e.offering.semester = self.currentSemester)
          ->collect(e | e.course.credits)
          ->sum()

      s = self.coreCredits

inv ConstraintName:
  let s : Integer =
    self.Enrolment

      ->collect(e | e.offering.credits)
      ->sum()

    v < 30
